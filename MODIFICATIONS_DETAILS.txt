APRÈS (détection type media):
  function openLb(n){
    ci=n;
    const lbImg = document.getElementById('lb-img');
    const lbVideo = document.getElementById('lb-video');
    const item = items[n];
    
    // Vérifier si c'est une vidéo ou une image
    if(item.type === 'video'){
      lbImg.style.display = 'none';
      lbVideo.style.display = 'block';
      lbVideo.src = item.s;
    } else {
      lbVideo.style.display = 'none';
      lbImg.style.display = 'block';
      lbImg.src = item.s;
    }
    
    document.getElementById('lb-t').textContent = item.t;
    document.getElementById('lb-s').textContent = item.d;
    document.getElementById('lb').classList.add('open');
  }

✓ Détection automatique du type (image vs vidéo)
